00484   /* This is special.  Within C++, you must specify `extern "C"' for */
00485   /* functions which are used via function pointers, and you also    */
00486   /* must do that for structures which contain function pointers to  */
00487   /* assure C linkage -- it's not possible to have (local) anonymous */
00488   /* functions which are accessed by (global) function pointers.     */
00489   /*                                                                 */

00491   /* FT_CALLBACK_DEF is used to _define_ a callback function.        */
00492   /*                                                                 */
00493   /* FT_CALLBACK_TABLE is used to _declare_ a constant variable that */
00494   /* contains pointers to callback functions.                        */
00495   /*                                                                 */

00497   /* that contains pointers to callback functions.                   */
